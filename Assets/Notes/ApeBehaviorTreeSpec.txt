Dead
  RESPAWN -> Moving(0)

Falling
  DEATH(Ape) -> Dead
  VOID       -> Dead

Jumping(Direction)
  DEATH(Ape) -> Dead
  FALL       -> Falling
  LAND       -> Moving(Velocity)

  Holding(Holdable)
    DEATH(Ape)      -> Dead
    FALL            -> Falling | Holdable.Free
    DEATH(Holdable) -> Jumping(Direction)

Moving(Velocity)
  DEATH(Ape)                  -> Dead
  FALL                        -> Falling | Holdable.Free
  JUMP(Velocity.normalized)   -> Jumping(Velocity.normalized)
  AIM(Direction,Duration > 0) -> Moving(Velocity)/Aiming(Direction,Duration)
  HOLD(Holdable)              -> Moving(Velocity)/Holding(Holdable)

  Holding(Holdable)
    DEATH(Ape)                -> Dead | Holdable.Free
    FALL                      -> Falling | Holdable.Free
    DEATH(Holdable)           -> Moving(Velocity)
    JUMP(Velocity.normalized) -> Jumping(Velocity.normalized)/Holding(Holdable)
    THROW                     -> Holdable.Thrown(Force)

  Aiming(Direction,Duration) 
    DEATH(Ape)             -> Dead
    FALL                   -> Falling
    TARGET(Targetable)     -> Moving(Velocity)/Aiming(Direction,Duration)/Targeting(Targetable)
    NO_AIM(Direction = 0)  -> Moving(Velocity)
    NO_METER(Duration = 0) -> Moving(Velocity)

    Targeting(Targetable)
      DEATH(Ape)                  -> Dead
      FALL                        -> Falling
      DEATH(Target)               -> Moving(Velocity)/Aiming(Direction,Duration)
      RANGE(Target > TargetRange) -> Moving(Velocity)/Aiming(Direction,Duration)
      NO_AIM(Direction = 0)       -> Moving(Velocity)
      NO_METER(Duration = 0)      -> Moving(Velocity)
      POUNCE                      -> Pouncing(Targetable)

Pouncing(Targetable,Remaining)
  DEATH(Ape)             -> Dead
  COLLIDE(Static)        -> Falling
  COLLIDE(Blocker)       -> Falling
  TIMEOUT(Remaining = 0) -> Falling
  ARRIVE                 -> Targetable.PounceOn Perched(Targetable)

Stomping(Targetable,Remaining)
  DEATH(Ape)             -> Dead
  COLLIDE(Static)        -> Falling
  COLLIDE(Blocker)       -> Falling
  TIMEOUT(Remaining = 0) -> Falling
  ARRIVE                 -> Targetable.StompOn | Perched(Targetable)

Perched(Perchable)
  DEATH(Ape)                  -> Dead
  DEATH(Perchable)            -> Falling
  JUMP(Direction)             -> Jumping(Direction)
  AIM(Direction,Duration > 0) -> Perched(Perchable)/Aiming(Direction,Duration)

  Aiming(Direction,Duration)
    DEATH(Ape)             -> Dead
    DEATH(Perchable)       -> Falling
    TARGET(Targetable)     -> Perched(Perchable)/Aiming(Direction,Duration)/Targeting(Targetable)
    NO_AIM(Direction = 0)  -> Perched(Perchable)
    NO_METER(Duration = 0) -> Perched(Perchable)

    Targeting(Targetable)
      DEATH(Ape)                      -> Dead
      DEATH(Targetable)               -> Moving(Velocity)/Aiming(Direction,Duration)
      RANGE(Targetable > TargetRange) -> Moving(Velocity)/Aiming(Direction,Duration)
      NO_AIM(Direction = 0)           -> Moving(Velocity)
      NO_METER(Duration = 0)          -> Moving(Velocity)
      POUNCE                          -> Pouncing(Targetable)
      HIT(Targetable2)                -> Targetable2.Hit(Direction) | Perched

      ChargingStomp(Duration)
        DEATH(Ape)                    -> Dead
        ABORT                         -> Perched(Perchable)/Aiming(Direction,Duration)/Targeting(Targetable)
        POUNCE(Duration < ChargeTime) -> Targetable.PounceFrom | Pouncing(Targetable)
        STOMP(Duration = ChargeTime)  -> Targetable.StompFrom | Stomping(Targetable)
